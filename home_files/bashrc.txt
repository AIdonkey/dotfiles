# BASHRC

# Default Editors
export EDITOR=sublime
export VISUAL=vim

# History
export HISTCONTROL=erasedups # Remove duplicate entries.
export HISTSIZE=1000 # Keep a lengthy history.
export HISTIGNORE="..:...:c:h:l:l1:p:pwd:gst:gd:exit:* --help" # Exclude mundane commands.

# Colors
. "$HOME/.bash_colors"

# Aliases
. "$HOME/.bash_aliases"

# Functions
. "$HOME/.bash_functions-private"
. "$HOME/.bash_functions-public"

# Command Prompt (http://jonisalonen.com/2012/your-bash-prompt-needs-this)
export GIT_PS1_SHOWDIRTYSTATE=true
export GIT_PS1_SHOWUNTRACKEDFILES=true
export GIT_PS1_SHOWSTASHSTATE=true
export GIT_PS1_SHOWUPSTREAM="auto"
separator="|"
prompt="[" # Opening bracket.
prompt+="\[$GREY\]\t\[$NORMAL\]" # Grey timestamp.
prompt+="$separator"
prompt+="\u@\h" # Default color user and host (i.e. user@host).
prompt+="$separator"
prompt+="\$(__git_ps1 \"\[$BLUE_BACKGROUND\]%s \$(gail)\[$NORMAL\]$separator\")" # Purple Git branch with separator.
prompt+="\[$CYAN\]\${PWD##*/}\[$NORMAL\]" # Cyan current directory.
prompt+="]> " # Closing bracket with prompt indicator.
export PS1=$prompt

# Bash Completion (http://bash-completion.alioth.debian.org)
if [[ -f $(brew --prefix)/etc/bash_completion ]]; then
  . $(brew --prefix)/etc/bash_completion
fi

# Z (https://github.com/rupa/z)
. `brew --prefix`/etc/profile.d/z.sh

# rbenv (https://github.com/sstephenson/rbenv)
if [[ -d /usr/local/bin ]]; then
  export PATH="/usr/local/bin:$PATH" # OSX
fi

if [[ -d $HOME/.rbenv/bin ]]; then
  export PATH="$HOME/.rbenv/bin:$PATH" # Ubuntu
fi

eval "$(rbenv init -)"

# NPM (http://nodejs.org)
export PATH="$PATH:/usr/local/share/npm/bin"

# Travis CI (https://travis-ci.org)
if [[ -e "$HOME/.travis/travis.sh" ]]; then
  source "$HOME/.travis/travis.sh"
fi
