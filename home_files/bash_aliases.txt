# BASH ALIASES

# Section: General
alias ..="cd .."
alias ...="cd ../.."
alias c="clear"
alias h="history"
alias l="ls -alh"
alias l1='ls -A1 | pbcopy && echo -e "`pbpaste` \n(copied to clipboard)"'
alias p='pwd | tr -d "\r\n" | pbcopy && echo "`pbpaste` (copied to clipboard)"'

alias home="cd $HOME"
alias bashs="exec $SHELL"

# Section: Network
alias ipa='curl -s checkip.dyndns.org | grep -Eo "[0-9\.]+" | pbcopy && echo "`pbpaste` (copied to clipboard)"'
